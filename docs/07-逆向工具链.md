## 7.3 浏览器控制

### browser-control.py

```python
class ReverseBrowser:
    """专为逆向设计的浏览器控制"""

    def __init__(self, target_url, stealth=True):
        self.target_url = target_url
        self.stealth = stealth
        self.browser = None

    def launch(self):
        """启动反检测浏览器"""
        options = ChromeOptions()
        if self.stealth:
            # 反检测配置
            options.add_argument("--disable-blink-features=AutomationDetected")
            options.add_argument("--disable-dev-shm-usage")
            options.add_argument("--disable-infobars")
            options.add_experimental_option("excludeSwitches", ["enable-automation"])
            # 注入 stealth 脚本
        self.browser = Chrome(options=options)
        self._inject_stealth()

    def _inject_stealth(self):
        """注入反检测脚本"""
        stealth_js = Path("tools/browsers/stealth/stealth.min.js").read_text()
        self.browser.execute_script(stealth_js)

    def inject_debugger_bypass(self):
        """过反调试"""
        bypass_code = Path("tools/browsers/stealth/bypass-debugger.js").read_text()
        self.browser.execute_script(bypass_code)

    def hook_console_crypto(self):
        """Hook console.debug，找到加密入口"""
        hook = """

                    stack: new Error().stack,
```

```python
# tools/mitm-script.py
from mitmproxy import http
import json

class ReverseCapture:
    def __init__(self, target_patterns):
        self.target_patterns = target_patterns  # ["api.example.com", "/sign"]
        self.requests = []
        self.responses = []

    def request(self, flow: http.HTTPFlow):
        if self._is_target(flow):
            self.requests.append({
                "url": flow.request.url,
                "method": flow.request.method,
                "headers": dict(flow.request.headers),
                "content": flow.request.content,
                "timestamp": flow.request.timestamp_start
            })

    def response(self, flow: http.HTTPFlow):
        if self._is_target(flow):
            self.responses.append({
                "url": flow.request.url,
                "status": flow.response.status_code,
                "headers": dict(flow.response.headers),
                "content": flow.response.content,
                "timestamp": flow.response.timestamp_start
            })

    def _is_target(self, flow):
        return any(p in flow.request.url for p in self.target_patterns)

    def export_har(self, path):
        with open(path, "w") as f:
            json.dump({
                "requests": self.requests,
                "responses": self.responses
            }, f, indent=2, default=str)

addons = [ReverseCapture(["example.com", "/api/", "/sign"])]
```

```javascript
// tools/frida/scripts/ssl-unpinning.js
// 适用: 大多数 Android App

Java.perform(function() {
    var TrustManager = Java.use('javax.net.ssl.TrustManager');
    var SSLContext = Java.use('javax.net.ssl.SSLContext');

    // 绕过 Android 的 SSL Pinning
    TrustManager.implementation.createTrustManager = function(type) {
        return TrustManager.$new();
    };

    // Hook 网络请求，打印加密参数
    var RequestUtil = Java.use('com.example.app.utils.RequestUtil');
    RequestUtil.getSign.implementation = function(a, b) {
        var sign = this.getSign(a, b);
        console.log('[SIGN] a=' + a + ', b=' + b + ', sign=' + sign);
        return sign;
    };
});
```

### AES 签名提取

```javascript
// tools/frida/scripts/cryptojs-aes-hook.js

Java.perform(function() {
    var AES = Java.use('javax.crypto.spec.SecretKeySpec');

    AES.$init.overload('[B', 'java.lang.String').implementation = function(key, algorithm) {
        var keyHex = Java.use('android.util.Base64').encodeToString(key, 2);
        console.log('[AES Init] Key=' + keyHex + ', Algorithm=' + algorithm);
        return this.$init(key, algorithm);
    };

    var Cipher = Java.use('javax.crypto.Cipher');
    Cipher.doFinal.overload('[B').implementation = function(data) {
        var dataHex = Java.use('android.util.Base64').encodeToString(data, 2);
        console.log('[AES doFinal] Data=' + dataHex);
        return this.doFinal(data);
    };
});
```

```python
# tools/python-utils/crypto-utils/crypto_common.py

import hashlib
import hmac
import base64
from Crypto.Cipher import AES, DES, DES3, PKCS1_v1_5
from Crypto.PublicKey import RSA
import rsa

ALGORITHMS = {
    "aes-256-cbc": AES_cbc,
    "aes-256-ecb": AES_ecb,
    "des": DES,
    "3des": DES3,
    "rsa": RSA,
    "hmac-sha256": hmac_sha256,
    "md5": md5,
}

def detect_algorithm(encrypted_hex, key=None):
    """自动检测加密算法（启发式）"""
    data = bytes.fromhex(encrypted_hex) if isinstance(encrypted_hex, str) else encrypted_hex
    length = len(data)

    if length == 16: candidates = ["aes-128", "des", "md5"]
    elif length == 24: candidates = ["3des"]
    elif length == 32: candidates = ["aes-256", "sha256"]
    elif length % 16 == 0: candidates = ["aes-*"]
    else: candidates = ["rsa", "unknown"]

    return candidates

def verify_signature(script, test_cases):
    """验证生成的签名脚本"""
    results = []
    for tc in test_cases:
        got = script.sign(tc["input"])
        results.append({
            "input": tc["input"],
            "expected": tc["expected"],
            "got": got,
            "match": got == tc["expected"]
        })
    return results
```

## 7.7 工具状态检查

```python
# scripts/check-tools.py

TOOLS = {
    "python": ["python3", "--version"],
    "node": ["node", "--version"],
    "mitmproxy": ["mitmproxy", "--version"],
    "frida": ["frida", "--version"],
    "frida-server": ["frida-ps", "-U"],
    "apktool": ["java", "-jar", "tools/apktool.jar", "--version"],
    "jadx": ["tools/jadx/bin/jadx", "--version"],
    "chrome": ["google-chrome", "--version"],
}

def check_all():
    status = {}
    for name, cmd in TOOLS.items():
        try:
            result = subprocess.run(cmd, capture_output=True, text=True)
            version = result.stdout.strip().split("\n")[0]
            status[name] = {"ok": True, "version": version}
        except Exception as e:
            status[name] = {"ok": False, "error": str(e)}
    return status
```
